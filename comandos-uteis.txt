[Comandos de Terminal]

# Criar um projeto Django
django-admin startproject <nome_do_projeto>

# Criar uma nova app para o projeto Django
python manage.py startapp

# Rodar o projeto
python manage.py runserver

# Entrar na pasta
cd <nome_da_pasta>
